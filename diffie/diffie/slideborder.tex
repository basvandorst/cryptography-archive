\macromode
\def\botbordergap#1{\raise .5\borderthick \crush{\vss \hbox{#1} \vss}}
\def\topbordergap#1{\lower .5\borderthick \crush{\vss \hbox{#1} \vss}}

\def\botrule#1
    {\leaders\hrule height \borderthick depth 0pt\hskip 0pt plus #1}

\def\leftrule{\vrule width \borderthick}

\def\rightrule{\vrule width \borderthick}

\def\slideborder#1
   {\vbox to \borderheight{{\lineskip 0pt\baselineskip 0pt
                            \slidehead \slidebody{#1} \slidefoot}}}

\def\slidehead
   {\vbox{\crush{\baselineskip0pt\lineskip0pt
                 \vss
                 \hbox to \borderwidth
                         {\toplefttag \hfill \topcenttag \hfill \toprighttag}
                 \vskip \tagraise}
          \hbox to \borderwidth
                  {\topleft\toprule{40pt}
                   \topbordergap{\topleftgap}\toprule{60pt}
                   \topbordergap{\topcentgap}\toprule{60pt}
                   \topbordergap{\toprightgap}\toprule{40pt}
                   \topright}}}

\def\slidebody#1
   {\hbox to \borderwidth
         {\leftrule \hss
          \setbox4=\vbox{\hbox{\topleft}\hbox{\botleft}}
          \setbox4=\hbox{\hskip \borderheight \hskip -\ht4}
          \vbox to \wd4{\vss #1 \vss}
          \hss \rightrule}}

\def\slidefoot
   {\vbox{\hbox to \borderwidth
                  {\botleft\botrule{20pt}
                   \botbordergap{\botleftgap}\botrule{80pt}
                   \botbordergap{\botcentgap}\botrule{80pt}
                   \botbordergap{\botrightgap}\botrule{20pt}
                   \botright}}
          \crush{\baselineskip0pt\lineskip0pt
                 \vskip \taglower
                 \hbox to \borderwidth
                         {\botlefttag \hfill \botcenttag \hfill \botrighttag}
                 \vss}}

\def\slidenum{{\small \the\slidecount.}}

\def\toprule#1
 {\leaders\hrule height \borderthick depth 0pt \hskip 0pt plus #1}

\def\leftlabel#1{\def\ll{\lower.5ex\hbox{\ #1\ }}}
\def\rightlabel#1{\def\rl{\lower.5ex\hbox{\ #1\ }}}
\def\bottomlabel#1{\def\bl{\lower.5ex\hbox{\ #1\ }}}

\leftlabel{}  % Default left label is null
\rightlabel{} % Default right label is null
\bottomlabel{} % Default bottom label is null

\def\topleftgap{\ll} \def\topcentgap{} \def\toprightgap{\rl}
\def\botleftgap{\bl} \def\botcentgap{} \def\botrightgap{\slidelogo}

\def\toplefttag{} \def\topcenttag{} \def\toprighttag{}
\def\botlefttag{} \def\botcenttag{} \def\botrighttag{\slidenum}

\def\tagraise{\borderthick} \def\taglower{\borderthick}

\def\topleft{} \def\topright{} \def\botleft{} \def\botright{}

\def\slidelogo{}

% Square corners depending on two parameters: \borderthick and \cornerheight

\def\cornerheight{\borderhtick}

\def\fancysquarecorners{
 \gdef\topleft{\lower\cornerheight
                   \fourframe{\trim{}{\cornerheight}{\cornerheight}}
                             {-\borderthick}{0pt}{-\borderthick}{0pt}}
 \gdef\topright{\lower\cornerheight
                    \fourframe{\trim{}{\cornerheight}{\cornerheight}}
                              {-\borderthick}{0pt}{0pt}{-\borderthick}}
 \gdef\botleft{\fourframe{\trim{}{\cornerheight}{\cornerheight}}
                       {0pt}{-\borderthick}{-\borderthick}{0pt}}
 \gdef\botright{\fourframe{\trim{}{\cornerheight}{\cornerheight}}
                       {0pt}{-\borderthick}{0pt}{-\borderthick}}}

% Simpler square corners depending on: \borderthick

\def\corner{\frame{\trim{}{\borderthick}{\borderthick}}{-.5\borderthick}}

\def\squarecorners
 {\gdef\topleft{\lower\borderthick\corner}
  \gdef\topright{\lower\borderthick\corner}
  \gdef\botleft{\corner} \gdef\botright{\corner}}

\def\squareborder#1
    {\vbox{\frame{\trim{#1}{\borderheight}{\borderwidth}}{\borderthick}
           \hbox to \borderwidth{\hfill \the\slidecount. \quad}}}

% Normal round courners

\def\roundcorners
 {\gdef\topleft  {{\corners A}}
  \gdef\topright {{\corners B}}
  \gdef\botleft  {{\corners C}}
  \gdef\botright {{\corners D}}}
